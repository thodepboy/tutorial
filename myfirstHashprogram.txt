
#include <bits/stdc++.h>
using namespace std;

class Cell
{
public:
    int key;
    Cell *next;
    Cell(int k)
    {
        key = k;
        next = NULL;
    }
    friend class Hashtable;
};
class Hashtable
{
private:
    int m;
    Cell **T;
public:
    Hashtable(int size = 100)
    {
        m= size;
        T = new Cell*[m];
        for(int i=0;i<m;i++)
            T[i]=NULL;
    }
    int Getslot(int k)
    {
        return k%m;
    }
    void Insert(int k)
    {
        int j =Getslot(k);
        if(T[j] == NULL)
        {
            T[j] = new Cell(k);
        }
        else
        {
            Cell *x = new Cell(k);
            Cell *cur = T[j];
            while (cur->next)
            {
                cur= cur->next;
            }
            cur->next= x;
        }
    }
    bool Search(int k)
    {
        int j = Getslot(k);
        for(Cell *cur = T[j]; cur != NULL; cur = cur->next)
        {
            if(cur->key == k)
                return true;
        }
        return false;
    }
    void Print()
    {
        for(int i=0;i<m;i++)
        {
            if(T[i] != NULL)
            {
                cout<<"hashBucket "<<i<<": ";
                for(Cell *cur = T[i]; cur!= NULL ;cur = cur->next)
                {
                    cout<<cur->key<<" ";
                }
                cout<<endl;
            }
        }
    }
    ~Hashtable() // thu dá»n table
    {
        for(int i=0;i<m;i++)
            if(T[i]!= NULL)
        {
            for(Cell *x = T[i];x != NULL;)
            {
                Cell *y=x;
                x= x->next;
                delete y;
            }
        }
    delete[] T;
    }
};
int main()
{
    Hashtable myFirstHash;
    int a;
    int b;
    cout<<"Nhap vao so input "<<endl;
    cin>>b;
    cout<<"Nhap vao gia tri: "<<end;
    while(b)
    {
        cin>>a;
        myFirstHash.Insert(a);
        b--;
    }
    if(myFirstHash.Search(36))
    cout<<"Found! "<<endl;
    else
        cout<<"Not Found! "<<endl;
    cout<<"___________________________________________________"<<endl;
    myFirstHash.Print(); //test for git
}
